from tkinter.constants import COMMAND
import speedtest
import tkinter as tk
from tkinter.ttk import *
from ttkthemes import *
from threading import Thread
import ctypes
from tkinter import messagebox


def speed_now():
    speed = speedtest.Speedtest()
    speed.get_servers()
    dl = speed.download()
    ul = speed.upload()
    dl = f"下载速度:{dl/1024/1024:.2f}MB"
    ul = f"上传速度:{ul/1024/1021:.2f}MB"
    result = {"dl": dl, "ul": ul}
    return result


window = ThemedTk(theme='adapta', toplevel=True, themebg=True)
ctypes.windll.shcore.SetProcessDpiAwareness(1)
ScaleFactor = ctypes.windll.shcore.GetScaleFactorForDevice(0)
window.tk.call('tk', 'scaling', ScaleFactor/75)
window.title("网络测速")
window.geometry("400x300")

grop = Labelframe(window).pack()
l1 = Label(master=window, text='测速工具', font=('楷体', 20))
l1.pack(padx=1, pady=2)
l2 = Label(grop, text='下载速度:__MB', font=('楷体', 15))
l3 = Label(grop, text='上传速度:__MB', font=('楷体', 15))
l2.pack()
l3.pack()


def task():
    result = speed_now()
    l2.config(text=result.get('dl'))
    l3.config(text=result.get('ul'))


def test():
    def show_prog():
        top = tk.Toplevel()
        top.attributes("-toolwindow", 1)
        top.wm_attributes("-topmost", 1)
        top.title("进度条")
        Label(top, text="正在查询，时间有点长，请稍等片刻", font=('楷体', 18)).pack()
        prog = Progressbar(top, mode="indeterminate", maximum=10, length=200)
        prog.pack()
        prog.start()
        task()
        prog.stop()
        top.destroy()
        msg = messagebox.showinfo(title='注意', message='查询成功！')
    t = Thread(target=show_prog)
    t.start()


b1 = Button(window, text='开始', command=test).pack()
l4 = Label(window, text='作者:sunshinehyc')
l5 = Label(window, text='联系:sunshine_hyc@outlook.com')
l4.pack()
l5.pack()


window.mainloop()
